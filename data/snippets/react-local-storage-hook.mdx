This custom hook provides a clean interface for working with localStorage in React applications.

## Usage

```javascript
import { useLocalStorage } from './useLocalStorage'

function MyComponent() {
  const [value, setValue] = useLocalStorage('myKey', 'defaultValue')
  
  return (
    <div>
      <input 
        value={value} 
        onChange={(e) => setValue(e.target.value)} 
      />
    </div>
  )
}
```

## Implementation

```javascript
import { useState, useEffect } from 'react'

function _readLocalStorage(key, initialValue) {
  if (typeof window === 'undefined') return initialValue
  try {
    const item = window.localStorage.getItem(key)
    return item ? JSON.parse(item) : initialValue
  } catch (err) {
    console.warn('useLocalStorage: failed to read', key, err)
    return initialValue
  }
}

export function useLocalStorage(key, initialValue) {
  // Use a safe reader so this hook won't throw during SSR
  const [storedValue, setStoredValue] = useState(() => _readLocalStorage(key, initialValue))

  // Keep state up-to-date if value exists in localStorage and component mounts in browser
  useEffect(() => {
    if (typeof window === 'undefined') return
    try {
      const item = window.localStorage.getItem(key)
      if (item) setStoredValue(JSON.parse(item))
    } catch {}
  }, [key])

  const setValue = (value) => {
    try {
      const valueToStore = value instanceof Function ? value(storedValue) : value
      setStoredValue(valueToStore)
      if (typeof window !== 'undefined') {
        window.localStorage.setItem(key, JSON.stringify(valueToStore))
      }
    } catch (err) {
      console.warn('useLocalStorage: failed to set', key, err)
    }
  }

  return [storedValue, setValue]
}
```

## Features

- Automatic JSON serialization/deserialization
- Error handling for localStorage access
- Same API as useState
- TypeScript support
- SSR safe (with Next.js)
